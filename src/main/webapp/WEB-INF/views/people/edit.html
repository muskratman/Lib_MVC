CREATE TABLE IF NOT EXISTS Book( book_id int GENERATED BY DEFAULT AS IDENTITY, title varchar(255), author varchar(255), year_of_publishing int CHECK (year_of_publishing > 1600), person_id int REFERENCES person(person_id) ON DELETE SET NULL, PRIMARY KEY(title, author, year_of_publishing)) CREATE TABLE IF NOT EXISTS Person( person_id int GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY, fio varchar(255) UNIQUE, year_of_birth int CHECK (year_of_birth > 1900))
